Wallet Verification

We use cryptographic signatures to verify the ownership of candidate wallets, ensuring the authenticity of their applications and linked credentials.

// Example JavaScript snippet for wallet verification
import { ethers } from 'ethers';

async function verifySignature(address, signature, message) {
  try {
    // Recover the address of the signer
    const signer = ethers.verifyMessage(message, signature);
    
    // Check if the recovered address matches the claimed address
    return signer.toLowerCase() === address.toLowerCase();
  } catch (error) {
    console.error("Signature verification failed:", error);
    return false;
  }
}

// Example usage in application flow
async function handleApplicationSubmit(candidateData) {
  const { walletAddress, signature, message } = candidateData;
  
  // First verify wallet ownership
  const isVerified = await verifySignature(walletAddress, signature, message);
  
  if (!isVerified) {
    throw new Error("Wallet verification failed");
  }
  
  // Proceed with application submission if verified
  return submitApplication(candidateData);
}

Smart Contract Integration

Our platform integrates with smart contracts to manage job postings, applications, and credential verification in a fully decentralized manner.

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.17;

import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Counters.sol";

contract FarcasterRecruit is Ownable {
    using Counters for Counters.Counter;
    
    Counters.Counter private _jobIds;
    Counters.Counter private _applicationIds;
    
    struct Job {
        uint256 id;
        address recruiter;
        string title;
        string description;
        string[] requiredSkills;
        uint256 compensation;
        bool isTokenCompensated;
        bool isRemote;
        bool isActive;
        uint256 createdAt;
    }
    
    struct Application {
        uint256 id;
        uint256 jobId;
        address candidate;
        string farcasterID;
        string credentialsHash; // IPFS hash to credentials JSON
        string[] skills;
        ApplicationStatus status;
        uint256 appliedAt;
    }
    
    enum ApplicationStatus { Applied, Reviewing, Rejected, Hired }
    
    mapping(uint256 => Job) public jobs;
    mapping(uint256 => Application) public applications;
    mapping(address => uint256[]) public recruiterJobs;
    mapping(address => uint256[]) public candidateApplications;
    
    event JobCreated(uint256 indexed jobId, address indexed recruiter, string title);
    event ApplicationSubmitted(uint256 indexed applicationId, uint256 indexed jobId, address indexed candidate);
    event ApplicationStatusChanged(uint256 indexed applicationId, ApplicationStatus status);
    
    // Create a new job posting
    function createJob(
        string memory title,
        string memory description,
        string[] memory requiredSkills,
        uint256 compensation,
        bool isTokenCompensated,
        bool isRemote
    ) external returns (uint256) {
        _jobIds.increment();
        uint256 jobId = _jobIds.current();
        
        jobs[jobId] = Job({
            id: jobId,
            recruiter: msg.sender,
            title: title,
            description: description,
            requiredSkills: requiredSkills,
            compensation: compensation,
            isTokenCompensated: isTokenCompensated,
            isRemote: isRemote,
            isActive: true,
            createdAt: block.timestamp
        });
        
        recruiterJobs[msg.sender].push(jobId);
        
        emit JobCreated(jobId, msg.sender, title);
        
        return jobId;
    }
    
    // Submit application for a job
    function applyForJob(
        uint256 jobId,
        string memory farcasterID,
        string memory credentialsHash,
        string[] memory skills
    ) external returns (uint256) {
        require(jobs[jobId].isActive, "Job is not active");
        
        _applicationIds.increment();
        uint256 applicationId = _applicationIds.current();
        
        applications[applicationId] = Application({
            id: applicationId,
            jobId: jobId,
            candidate: msg.sender,
            farcasterID: farcasterID,
            credentialsHash: credentialsHash,
            skills: skills,
            status: ApplicationStatus.Applied,
            appliedAt: block.timestamp
        });
        
        candidateApplications[msg.sender].push(applicationId);
        
        emit ApplicationSubmitted(applicationId, jobId, msg.sender);
        
        return applicationId;
    }
    
    // Update application status (recruiter only)
    function updateApplicationStatus(uint256 applicationId, ApplicationStatus newStatus) external {
        uint256 jobId = applications[applicationId].jobId;
        require(jobs[jobId].recruiter == msg.sender, "Not authorized");
        
        applications[applicationId].status = newStatus;
        
        emit ApplicationStatusChanged(applicationId, newStatus);
    }
    
    // Get jobs created by a recruiter
    function getRecruiterJobs(address recruiter) external view returns (uint256[] memory) {
        return recruiterJobs[recruiter];
    }
    
    // Get applications submitted by a candidate
    function getCandidateApplications(address candidate) external view returns (uint256[] memory) {
        return candidateApplications[candidate];
    }
}

API Integration Example

Our backend service provides RESTful APIs to interact with the Farcaster Recruit platform and on-chain data.

// Sample API integration for fetching on-chain credentials

// Fetch candidate's on-chain credentials
async function fetchOnChainCredentials(walletAddress) {
  try {
    const response = await fetch(`/api/credentials/${walletAddress}`);
    
    if (!response.ok) {
      throw new Error(`Error: ${response.status}`);
    }
    
    const credentials = await response.json();
    return credentials;
  } catch (error) {
    console.error("Error fetching credentials:", error);
    throw error;
  }
}

// Verify GitHub contributions on-chain
async function verifyGitHubContributions(githubHandle, walletAddress) {
  try {
    const response = await fetch('/api/verify/github', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        githubHandle,
        walletAddress
      }),
    });
    
    if (!response.ok) {
      throw new Error(`Error: ${response.status}`);
    }
    
    const result = await response.json();
    return result.verified;
  } catch (error) {
    console.error("Error verifying GitHub:", error);
    return false;
  }
}

// Submit job application with credentials
async function submitApplication(jobId, candidateData, signature) {
  try {
    const response = await fetch('/api/applications/submit', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        jobId,
        candidateData,
        signature
      }),
    });
    
    if (!response.ok) {
      throw new Error(`Error: ${response.status}`);
    }
    
    const result = await response.json();
    return result;
  } catch (error) {
    console.error("Error submitting application:", error);
    throw error;
  }
}
